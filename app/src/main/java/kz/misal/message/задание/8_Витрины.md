# UI Catalog

**Cтрогий отраслевой стандарт.**

В профессиональной разработке такие экраны-витрины называют **UI Catalog** (Каталог интерфейса) или **Living UI Kit** (Живой UI-кит).

Дизайн-система не существует на бумаге или в Figma — она существует только тогда, когда работает в коде. 
Создание витрин для цветов, форм и отступов решает три гигантские проблемы:
1. **Тестирование темной темы:** Вы мгновенно видите, если какой-то цвет "режет глаз" ночью или если текст сливается с фоном.
2. **Проверка кастомных элементов:** Вы можете убедиться, что ваши `CustomColors` и `AppSpacing` правильно подтянулись из `CompositionLocal`.
3. **Онбординг новичков:** Новый разработчик открывает один файл и сразу видит все "детали лего", из которых предстоит собирать экраны.

Вот как пишутся профессиональные витрины для остальных элементов системы.

### 1. Витрина цветов (Colors Preview)

Главная фишка цветовой витрины — она всегда запускается **в двух экземплярах**: для светлой и темной темы. Мы создаем удобную плашку, которая показывает сам цвет и цвет текста на нем (пару `color` / `onColor`).

```kotlin
import android.content.res.Configuration
import androidx.compose.foundation.background
import androidx.compose.foundation.layout.*
import androidx.compose.foundation.lazy.grid.GridCells
import androidx.compose.foundation.lazy.grid.LazyVerticalGrid
import androidx.compose.material3.*
import androidx.compose.runtime.Composable
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.graphics.Color
import androidx.compose.ui.tooling.preview.Preview
import androidx.compose.ui.unit.dp

// 1. Вспомогательный компонент для одной цветовой плашки
@Composable
private fun ColorSwatch(name: String, backgroundColor: Color, textColor: Color) {
    Box(
        modifier = Modifier
            .fillMaxWidth()
            .height(80.dp)
            .background(backgroundColor),
        contentAlignment = Alignment.Center
    ) {
        Text(
            text = name,
            style = MaterialTheme.typography.labelMedium,
            color = textColor
        )
    }
}

// 2. Сам экран витрины
@Composable
fun ColorSystemScreen() {
    LazyVerticalGrid(
        columns = GridCells.Fixed(2), // В две колонки
        modifier = Modifier.fillMaxSize().background(MaterialTheme.colorScheme.background)
    ) {
        // Базовые цвета M3
        item { ColorSwatch("Primary", MaterialTheme.colorScheme.primary, MaterialTheme.colorScheme.onPrimary) }
        item { ColorSwatch("Secondary", MaterialTheme.colorScheme.secondary, MaterialTheme.colorScheme.onSecondary) }
        item { ColorSwatch("Surface", MaterialTheme.colorScheme.surface, MaterialTheme.colorScheme.onSurface) }
        item { ColorSwatch("Surface Variant", MaterialTheme.colorScheme.surfaceVariant, MaterialTheme.colorScheme.onSurfaceVariant) }
        item { ColorSwatch("Error", MaterialTheme.colorScheme.error, MaterialTheme.colorScheme.onError) }
        
        // Наши кастомные цвета!
        item { 
            ColorSwatch(
                name = "Custom Success", 
                backgroundColor = MaterialTheme.customColors.success, 
                textColor = MaterialTheme.customColors.onSuccess
            ) 
        }
    }
}

// 3. Запускаем ДВА превью: для светлой и темной темы
@Preview(name = "Light Colors", showBackground = true)
@Composable
fun ColorsPreviewLight() {
    AppTheme(darkTheme = false) { ColorSystemScreen() }
}

@Preview(name = "Dark Colors", uiMode = Configuration.UI_MODE_NIGHT_YES, showBackground = true)
@Composable
fun ColorsPreviewDark() {
    AppTheme(darkTheme = true) { ColorSystemScreen() }
}
```

### 2. Витрина Форм и Отступов (Shapes & Spacing Preview)

Эту витрину часто объединяют. Задача — показать, как работает 8-пиксельная сетка (наш `AppSpacing`) и как выглядят радиусы скругления (`MaterialTheme.shapes`).

```kotlin
import androidx.compose.foundation.layout.*
import androidx.compose.foundation.shape.CornerSize
import androidx.compose.material3.*
import androidx.compose.runtime.Composable
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.tooling.preview.Preview
import androidx.compose.ui.unit.dp

@Composable
private fun ShapeSwatch(name: String, shape: androidx.compose.ui.graphics.Shape) {
    Surface(
        shape = shape,
        color = MaterialTheme.colorScheme.primary,
        modifier = Modifier.size(80.dp)
    ) {
        Box(contentAlignment = Alignment.Center) {
            Text(text = name, color = MaterialTheme.colorScheme.onPrimary, style = MaterialTheme.typography.labelMedium)
        }
    }
}

@Preview(showBackground = true)
@Composable
fun ShapesAndSpacingPreview() {
    AppTheme {
        Surface(modifier = Modifier.fillMaxSize(), color = MaterialTheme.colorScheme.background) {
            Column(
                // ИСПОЛЬЗУЕМ НАШИ ОТСТУПЫ!
                // Внешний отступ экрана = 24.dp
                modifier = Modifier.padding(MaterialTheme.spacing.large), 
                // Расстояние между рядами = 16.dp
                verticalArrangement = Arrangement.spacedBy(MaterialTheme.spacing.medium) 
            ) {
                Text("Формы (Shapes)", style = MaterialTheme.typography.titleLarge)
                
                Row(
                    // Расстояние между квадратиками = 8.dp
                    horizontalArrangement = Arrangement.spacedBy(MaterialTheme.spacing.small)
                ) {
                    ShapeSwatch("Small", MaterialTheme.shapes.small)
                    ShapeSwatch("Medium", MaterialTheme.shapes.medium)
                    ShapeSwatch("Large", MaterialTheme.shapes.large)
                }

                Spacer(modifier = Modifier.height(MaterialTheme.spacing.extraLarge))
                
                Text("Отступы (Spacing)", style = MaterialTheme.typography.titleLarge)
                Text("Расстояние между этими строками = extraSmall (4.dp)")
                // ... и так далее
            }
        }
    }
}
```

### Как это используется в команде

Обычно разработчики создают отдельный скрытый экран в приложении 
(например, по долгому тапу на номер версии в настройках для дебаг-сборок). 
Открывается полноценный "Каталог UI", где тестировщик (QA) или дизайнер может потыкать все кнопки во всех состояниях, 
посмотреть все цвета и шрифты прямо на живом телефоне.
